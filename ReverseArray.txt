#include <iostream>

using namespace std;

// Code for reversing elements of the array

int reverseArray(int arr[], int n){
    int s = 0;
    int e = n-1;
    
    
    
    while(s<e){
        swap(arr[s],arr[e]);
        s+=1;
        e-=1;
    }
    
    return 0;
}



int main()
{
    int arr[] = {10,20,30,40,45,60,70,89};
    
    int n = sizeof(arr)/sizeof(int);
    
    reverseArray(arr,n);
    for(int i=0;i<=n;i++){
        cout<<arr[i]<<" ";
    }
    cout<<endl;
    return 0;
}
