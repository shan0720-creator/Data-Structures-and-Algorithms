#include <iostream>

using namespace std;

int main()
{
    
     //  Arrays Input Output Update
     
     int marks[100];
     
     int n;
     cout<<"Enter the no of student";
     cin>>n;
     
     //Assign a value
     marks[0] = -1;
     
     // Input 
     for(int i = 1; i<=n; i++){  // Start Assigning marks from 1st place and go upto <=n
         cin>> marks[i];
         marks[i] = marks[i] *2;
     }
     
     for(int i=0;i<=n;i++){
         
         cout<< marks[i] << ",";
     }
     
     cout<<endl;
     
     
     // end of the program

    return 0;
}
